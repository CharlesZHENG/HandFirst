```
 class Program
    {
        static void Main(string[] args)
        {
            MyClass t = new MyClass();
            Task.Run(() => { t.Test(20); });
            Task.Run(() => { t.Test(30); });
            Console.WriteLine("Hello World!");
            Console.Read();
        }
    }

    public class MyClass
    {
        //lock是可重入锁
        private static object myLock = new object();
        public void Test(int i)
        {
            lock (myLock)
            {
                if (i > 0)
                {
                    i--;
                    Console.WriteLine(i);
                    Test(i);
                }
            }
        }
    }
```
